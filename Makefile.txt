# Database Commands
.PHONY: db-setup
db-setup:
	@echo "$(YELLOW)ðŸ—„ï¸  Setting up database...$(NC)"
	@$(DOCKER_COMPOSE) exec postgres psql -U postgres -c "CREATE DATABASE IF NOT EXISTS content_factory;"
	@$(PYTHON) database/init_database.py
	@make db-migrate
	@echo "$(GREEN)âœ“ Database setup completed$(NC)"

.PHONY: db-migrate
db-migrate:
	@echo "$(YELLOW)ðŸ”„ Running database migrations...$(NC)"
	@$(PYTHON) database/migrate.py
	@echo "$(GREEN)âœ“ Migrations completed$(NC)"

.PHONY: db-reset
db-reset:
	@echo "$(RED)âš ï¸  WARNING: This will delete all data!$(NC)"
	@read -p "Are you sure? [y/N] " -n 1 -r; \
	if [[ $$REPLY =~ ^[Yy]$$ ]]; then \
		echo ""; \
		echo "$(YELLOW)ðŸ—‘ï¸  Resetting database...$(NC)"; \
		$(DOCKER_COMPOSE) down -v; \
		$(DOCKER_COMPOSE) up -d postgres; \
		sleep 5; \
		make db-setup; \
		echo "$(GREEN)âœ“ Database reset completed$(NC)"; \
	else \
		echo ""; \
		echo "$(GREEN)âœ“ Operation cancelled$(NC)"; \
	fi

.PHONY: db-backup
db-backup:
	@echo "$(YELLOW)ðŸ’¾ Creating database backup...$(NC)"
	@mkdir -p backups
	@$(DOCKER_COMPOSE) exec postgres pg_dump -U postgres content_factory > backups/backup_$(shell date +%Y%m%d_%H%M%S).sql
	@echo "$(GREEN)âœ“ Backup created in backups/$(NC)"

# Cleanup
.PHONY: clean
clean:
	@echo "$(YELLOW)ðŸ§¹ Cleaning temporary files...$(NC)"
	@find . -type f -name "*.pyc" -delete
	@find . -type d -name "__pycache__" -delete
	@rm -rf .pytest_cache
	@rm -rf .coverage
	@rm -rf htmlcov
	@echo "$(GREEN)âœ“ Cleanup completed$(NC)"

.PHONY: clean-all
clean-all:
	@echo "$(RED)âš ï¸  This will remove all Docker images and volumes!$(NC)"
	@read -p "Are you sure? [y/N] " -n 1 -r; \
	if [[ $$REPLY =~ ^[Yy]$$ ]]; then \
		echo ""; \
		echo "$(YELLOW)ðŸ§¹ Deep cleaning...$(NC)"; \
		$(DOCKER_COMPOSE) down -v --rmi all; \
		docker system prune -af; \
		make clean; \
		echo "$(GREEN)âœ“ Deep cleanup completed$(NC)"; \
	else \
		echo ""; \
		echo "$(GREEN)âœ“ Cleanup cancelled$(NC)"; \
	fi
